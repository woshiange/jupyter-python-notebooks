Los Angeles Homicides

This dashboard provides insights into homicides that have occurred in Los Angeles City.

The data powering this dashboard is sourced from the Los Angeles Police Department (LAPD). For a detailed description of the dataset, visit Los Angeles Open DataPortal.

duckdb.sql("""
    create table la_homicide_data as
    select
        raw_data.*,
        area_mapping_df.zone,
        case
            when victime_descent_mapping_df.descent_name is null then 'Unknown'
            when victime_descent_mapping_df.descent_name in 
                    ('Other Asian', 'Chinese', 'Cambodian', 'Filipino', 'Japanese',
                    'Korean', 'Laotian', 'Vietnamese')
                then 'Asian'
            else victime_descent_mapping_df.descent_name
        end as descent_name
    from raw_data
    left outer join area_mapping_df
        on area_mapping_df.area_name = raw_data.area_name
    left outer join victime_descent_mapping_df
        on victime_descent_mapping_df.descent_code = vict_descent
    where
        crm_cd_desc = 'CRIMINAL HOMICIDE'
        and date_trunc('month', date_occ) < (
            select
                max(date_trunc('month', date_occ))
            from raw_data
        )
""")

evolution_df = duckdb.sql("""
        with
            base as (
                select
                    date_trunc('month', date_occ) as month_date,
                    count(*) as amount
                from la_homicide_data
                where
                    crm_cd_desc = 'CRIMINAL HOMICIDE'
                    and date_trunc('month', date_occ) >= (
                        select
                            max(date_trunc('month', date_occ)) - interval 12 month
                        from la_homicide_data
                    )
                group by
                    month_date
            )
        select
            monthname(month_date) || ' ' || year(month_date) as label,
            case
                when lag(amount) over (order by month_date) is null then amount
                else amount - lag(amount) over (order by month_date)
            end as amount
        from base
    """).df()

first_label = evolution_df['label'][0]


base_chart = alt.Chart(evolution_df).transform_window(
    window_sum_amount="sum(amount)",
    window_lead_label="lead(label)",
).transform_calculate(
    calc_lead="datum.window_lead_label === null ? datum.label : datum.window_lead_label",
    calc_prev_sum="datum.window_sum_amount - datum.amount",
    calc_amount="datum.amount",
    calc_text_amount=f"(datum.label !== '{first_label}' && datum.calc_amount > 0 ? '+' : '') + datum.calc_amount",
    calc_center="(datum.window_sum_amount + datum.calc_prev_sum) / 2",
    calc_sum_dec="datum.window_sum_amount < datum.calc_prev_sum ? datum.window_sum_amount : ''",
    calc_sum_inc="datum.window_sum_amount > datum.calc_prev_sum ? datum.window_sum_amount : ''",
).encode(
    x=alt.X(
        "label:O",
        axis=alt.Axis(title="", labelAngle=45),
        sort=None,
    )
)

color_coding = {
    "condition": [
        {"test": f"datum.label === '{first_label}'", "value": "#878d96"},

    "value": "#fa4d56",
}

bar = base_chart.mark_bar(size=45).encode(

victim_demographics_df = duckdb.sql("""
            with
                base_age as (
                    select
                        serie as age
                    from generate_series(5, 80, 5) as t(serie)
                ),
                base_gender as (
                    select 'Female' as gender
                    union
                    select 'Male' as gender
                ),
                base as (
                    select
                        age,
                        gender
                    from base_age, base_gender
                ),
                result as (
                    select
                        case
                            when vict_sex = 'M' then 'Male'
                            else 'Female'
                        end as gender,
                        cast(vict_age/5 as int)*5 as age,
                        count(*) as victims
                    from la_homicide_data
                    where
                        vict_sex is not null
                        and vict_sex in ('M', 'F')
                        and vict_age between 5 and 80
                        and vict_age is not null
                    group by
                        gender, age
                )
                select
                    base.age,
                    base.gender,
                    coalesce(result.victims, 0) as victims
                from base
                left outer join result
                    on base.gender = result.gender
                    and base.age = result.age

    """).df()

max_victim = max(victim_demographics_df['victims'])
max_victim_rounded = max_victim if max_victim % 10 == 0 else (int(max_victim/10) + 1)*10

zone_ranking_df = duckdb.sql("""
                with
                    months as (
                        select
                            distinct(date_trunc('month', date_occ)) as month_date
                        from la_homicide_data
                        where date_trunc('month', date_occ) > (
                            select
                                max(date_trunc('month', date_occ)) - interval 12 month
                            from la_homicide_data
                        )
                    ),
                    zones as (
                        select
                            distinct(zone) as zone
                        from la_homicide_data
                    ),
                    base as (
                        select *
                        from months, zones
                    ),
                    result as (
                        select
                            date_trunc('month', date_occ) as month_date,
                            zone,
                            count(*) as nb_victimes
                        from la_homicide_data
                        where crm_cd_desc = 'CRIMINAL HOMICIDE'
                        group by
                            month_date,
                            zone
                    )
                    select
                        base.month_date,
                        base.zone,
                        coalesce(result.nb_victimes, 0) as nb_victimes,
                        row_number() over (
                            partition by base.month_date
                            order by
                                coalesce(result.nb_victimes, 0) desc,
                                base.zone
                        ) as month_ranking
                    from base
                    left outer join result
                        on result.month_date = base.month_date
                        and result.zone = base.zone
                    order by
                        base.month_date,
                        month_ranking
    """).df()
